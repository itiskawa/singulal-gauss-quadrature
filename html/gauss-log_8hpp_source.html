240 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> chebyshev(std::size_t n,Eigen::Vector<T, Eigen::Dynamic> mom, Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> abj){
241 // size check
242 assert(2*n == mom.rows());
243 assert(2*n == abj.rows());
244
245 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> ab = Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>::Zero(n,2);
246
247 ab(0,0) = abj(0,0) + (mom[1]/mom[0]);
248 ab(0,1) = mom[0];
249
250 T s_1 = 0.;
251
252 // Computing sigma-matrix
253 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> sigma = Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>::Zero(n+1, 2*n);
254
255 // Initializing first row
256 for(int l = 0; l < 2*n; l++){
257 sigma(0, l) = mom[l];
258 }
259
260 // Filling in n following rows
261 for(int k = 1; k < n; k++){
262 for(int l = k; l < 2*n-k; l++){
263 if(k > 1){ s_1 = sigma(k-2, l); }
264 sigma(k,l) = sigma(k-1,l+1) - (ab(k-1,0) - abj(l,0))*sigma(k-1,l)-ab(k-1,1)*s_1 + abj(l,1)*sigma(k-1,l-1);
265
266 }
267 // alpha_k
268 ab(k, 0) = abj(k, 0)+ (sigma(k,k+1)/sigma(k,k)) - (sigma(k-1,k)/sigma(k-1,k-1));
269
270 // beta_k
271 ab(k, 1) = sigma(k,k)/sigma(k-1,k-1);
272 }
273 return ab;
274 }

281 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> nw(std::size_t n) {
282
283 // Finding the coefficients
284 double gamma_0 = 1; // given that a=b=0
285 Eigen::Vector<T, Eigen::Dynamic> mom = mmom_log(2*n);
286 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> abm = shifted_c_log(2*n);
287 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> ab = chebyshev(n, mom, abm);
288
289
290 // Solving for the coefficients with Golub-Welsch
291 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> J_n = this->tridiagCoeffs(ab, n);
292
293 // The self-adjoint solver. Significant speedup for large n
294 Eigen::SelfAdjointEigenSolver<Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>> solve(J_n); // yields much faster computations of high n
295
296 /* Extracting eigenvalues and vectors. Only the real part is considered as
297 * they are guaranteed to be real given the nature of the problem */
298 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> eigenvecs = solve.eigenvectors().real();
299 Eigen::Vector<T, Eigen::Dynamic> nodes= solve.eigenvalues().real();
300
301 // Solving the weights using Golub-Welsch algorithm formula
302 Eigen::Vector<T, Eigen::Dynamic> weights = Eigen::Vector<T, Eigen::Dynamic>::Zero(n);
303 for(int i = 0; i < n; i++){
304 weights[i] = gamma_0*pow(eigenvecs.col(i).normalized()[0], 2);
305 }
306
307 // Preparing return type
308 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> nw = Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>::Zero(n, 2);
309
310 nw.col(0) = nodes;
311 nw.col(1) = weights;
312
313 return nw;
314 }
