211 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> nw() {
212
213 double gamma_0 = gamma_zero();
214 std::size_t n = this->degree;
215
216 /* Computing the recurrence relation coefficients for the tridiagolanisation process
217 * Both functions below are from the abstract super-class template GaussRule<T>
218 */
219 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> coeffs = this->c_jacobi(n, this->alpha, this->beta);
220 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> J_n = this->tridiagCoeffs(coeffs, n);
221
222 // The self-adjoint solver. Significant speedup for large n
223 Eigen::SelfAdjointEigenSolver<Eigen::Matrix<T,Eigen::Dynamic, Eigen::Dynamic>> solve(J_n);
224
225 /* Extracting eigenvalues and vectors. Only the real part is considered as
226 * they are guaranteed to be real given the nature of the problem */
227 Eigen::Vector<T, Eigen::Dynamic> nodes = solve.eigenvalues().real();
228 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> eigenvecs = solve.eigenvectors().real();
229
230 // Solving the weights using Golub-Welsch algorithm formula
231 Eigen::Vector<T, Eigen::Dynamic> weights = Eigen::Vector<T, Eigen::Dynamic>::Zero(n);
232 for(int i = 0; i < n; i++){
233 weights[i] = gamma_0*pow(eigenvecs.col(i).normalized()[0], 2);
234 }
235
236 // Preparing more compact return type
237 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> nw = Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>::Zero(n, 2);
238 nw.col(0) = nodes;
239 nw.col(1) = weights;
240
241 return nw;
242
243 }
